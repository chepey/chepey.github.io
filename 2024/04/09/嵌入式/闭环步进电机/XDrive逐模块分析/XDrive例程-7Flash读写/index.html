#if (Stockpile_Page_Size != FLASH_PAGE_SIZE)	//和HAL库获取的Flash页大小比较,检查配置是否有效

结构体

typedef struct{
	//配置
	uint32_t	begin_add;			//起始地址
	uint32_t	area_size;			//区域大小
	uint32_t	page_num;				//芯片实体页数量
	//过程量
	uint32_t	asce_write_add;	//写地址
}Stockpile_FLASH_Typedef;

接口函数

void Stockpile_Flash_Data_Empty(Stockpile_FLASH_Typedef *stockpile)
  • 功能：Flash数据清空
  • 参数：
    • *stockpile：指向分区表实例的指针
  • 返回：
  • 实现：
    • 解锁Flash：HAL_FLASH_Unlock()
    • 循环遍历每一页
      • 创建初始化结构体
      • 擦除方式：页擦除
      • 擦除地址：起始位置+n*页大小
      • 擦除页数量：1
      • 调用擦除函数：HAL_FLASHEx_Erase(&erase_config, &page_error)
      • 等待最后一个操作完毕：FLASH_WaitForLastOperation(HAL_MAX_DELAY)
      • 清除FLASH_CR_PER比特
      • 锁定Flash

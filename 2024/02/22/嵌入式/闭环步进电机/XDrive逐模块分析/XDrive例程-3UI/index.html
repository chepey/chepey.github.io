类型定义

按钮状态类型

UI数据包，用于存放计时相关数据和标志

typedef struct{
	uint32_t		period_ms;				//周期
	uint32_t		flicker_timer_100ms;	//闪烁计时器
	uint32_t		flicker_timer_500ms;	//闪烁计时器
	bool			flicker_flag_100ms;		//闪烁标志
	bool			flicker_flag_500ms;		//闪烁标志
}REINui_Data_Typedef;

基本控件

typedef struct{
	//用户辨识
	uint8_t			*name;			//控件名称
	uint8_t			*icon_3232;		//图标(32*32)
	//拓扑
	void			*source_widget;	//源控件指针(指向来源控件) 
	void			*linked_widget;	//链表指针(指向同级别下一个控件)
	//绘制
	//动态刷新
	bool			update_drawing;											//绘制刷新
	void			(*drawing_function)(void* widget);	//绘制函数
	//高级功能包(由于C无法实现继承,故使用该方法实现在统一引用下定义不同数据)
	void			*feature_pack;		//高级功能包
}REINui_Widget_Typedef;
typedef struct{
	REINui_Widget_Typedef		*begin_widget;		//起始控件
	REINui_Widget_Typedef		*end_widget;		//结束控件
	uint32_t					duration_ms;		//过度时间
}REINui_Switches_Typedef;
typedef struct{
	//项目指示
	void		*target_widget;			//目标控件指针(指向列表的第一个单元)
	uint32_t	list_size;				//目录列表大小
	uint32_t	pitch_number;			//选中的项目编号(0开始)
	//过程变量(无需全局)
	REINui_Widget_Typedef		*up_widget;				//上项目
	REINui_Widget_Typedef		*middle_widget;			//中项目
	REINui_Widget_Typedef		*down_widget;			//下项目
}REINui_Directory_Typedef;

函数

内部服务函数

void REINui_Widget_Init(REINui_Widget_Typedef *_widget, uint8_t *_name, uint8_t *_icon_3232, void (*_drawing)(void*), void *_pack)
  • 功能：初始化控件，初始化对象
  • 参数：
    • *_widget：界面对象的指针
    • *_name：界面的名字
    • *_icon_3232：存放图标的指针
    • (*_drawing)(void*)：绘制函数
    • *_pack：高级功能包指针
  • 返回：
  • 实现：
    • 界面名称赋值
    • 上下拓扑结构均初始化为NULL
    • 为结构体链接图标指针
    • 为结构体链接绘制函数
    • 为结构体链接高级功能包（过渡、目录）
void REINui_Switches_Init(REINui_Switches_Typedef *_switches)
  • 功能：初始化高级包(过渡动画)
  • 参数：
    • *_switches：过渡结构体实例指针
  • 返回：
  • 实现：
    • 初始化前、后窗口对象为NULL
    • 初始化过渡时间为200ms
void REINui_Directory_Init(REINui_Directory_Typedef *_directory)
  • 功能：初始化高级包(目录)
  • 参数：
    • *_directory：目录结构体实例指针
  • 返回：
  • 实现：
    • 初始化列表第一个单元的指针为NULL
    • 初始化列表长度为0
    • 初始化当前选中窗口号为0
void REINui_Directory_Append_Widget(REINui_Widget_Typedef* _host, REINui_Widget_Typedef* _slave)
  • 功能：为目录添加子控件
  • 参数：
    • *_host：主窗口指针
    • * _slave：从窗口指针
  • 返回：
  • 实现：
    • 提取主窗口指针的高级功能包对象指针（指向目录解耦固体）为局部变量
    • 给主窗口的子一级的最后的位置添加从窗口
    • 将从窗口的上一级添加为_host（主窗口）
void REINui_Switches_drawing_function(void* _widget)
  • 功能：切换动画绘画函数
  • 参数：
  • 返回：
  • 实现：
    • 不做动画，直接跳转到目标界面
void REINui_Directory_drawing_function(void* _widget)
  • 功能：目录绘画函数
  • 参数：
    • *_widget：绘制目录的主界面指针
  • 返回：
  • 实现：
    • 响应刷新命令（若刷新命令=TRUE）
      • 关闭下一次刷新
      • 遍历列表获取列表大小
      • 获取目录表显示项目
      • 清除gram中数据
      • 绘制外框
      • 绘制位置指示
      • 绘制数据
      • 绘制选中图标
      • 刷新SSD1306
    • 响应用户输入
      • 处理向上移动（选中项目编号向上修改）
      • 处理向下移动（选中项目编号向下修改）
      • 处理向上长按（进入选中项目）
      • 处理向下长按（退回上一级）
void REINui_Desktop_drawing_function(void* _widget)
  • 功能：桌面绘画函数
  • 参数：
    • *_widget：界面指针
  • 返回：
  • 实现：
    • 桌面全局刷新
    • 清除gram中数据
    • 绘制模式行
    • 绘制关键状态行
    • 绘制不同模式下的特色指示
    • 刷新SSD1306
    • 用户输入响应
      • 向上短按（模式向上切换）
      • 向下短按（模式向下切换）
      • 向上长按（将界面模式下载到控制指令）
      • 向下长按（退回上一个项目）
